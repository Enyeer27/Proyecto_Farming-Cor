from flask import Flask, request, jsonify app = Flask(__name__) # Datos de ejemplo para frutas y verduras frutas = [ {"id": 1, "nombre": "Manzana", "cantidad": 10}, {"id": 2, "nombre": "Banana", "cantidad": 5}, ] verduras = [ {"id": 1, "nombre": "Zanahoria", "cantidad": 15}, {"id": 2, "nombre": "Lechuga", "cantidad": 20}, ] # Ruta para obtener la lista de frutas @app.route('/frutas', methods=['GET']) def obtener_frutas(): return jsonify(frutas) # Ruta para obtener la lista de verduras @app.route('/verduras', methods=['GET']) def obtener_verduras(): return jsonify(verduras) # Ruta para donar frutas @app.route('/donar/frutas', methods=['POST']) def donar_fruta(): data = request.get_json() nueva_fruta = { "id": len(frutas) + 1, "nombre": data["nombre"], "cantidad": data["cantidad"] } frutas.append(nueva_fruta) return jsonify(nueva_fruta), 201 # Ruta para donar verduras @app.route('/donar/verduras', methods=['POST']) def donar_verdura(): data = request.get_json() nueva_verdura = { "id": len(verduras) + 1, "nombre": data["nombre"], "cantidad": data["cantidad"] } verduras.append(nueva_verdura) return jsonify(nueva_verdura), 201 if __name__ == '__main__': app.run(debug=True)
